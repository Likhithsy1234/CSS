def isprimenum(n,i=1,c=0):
    if i*i>n:
         return c==2
    if n%i==0:
        c+=1
        if i!=n//i:
            c+=1
    return isprimenum(n,i+1,c)
data=int(input("enter a number"))
chk=isprimenum(data)
if chk==2:
    print(f"{data} is prime")

     print(f"{data}is prime")
